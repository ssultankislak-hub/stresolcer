<script>
const video = document.getElementById('v');
const waveCanvas = document.getElementById('waveCanvas');
const wctx = waveCanvas.getContext('2d');
const buffer = document.getElementById('buffer');
const bctx = buffer.getContext('2d', { willReadFrequently: true });
const btn = document.getElementById('btn');

let isActive = false;
let signal = [];
let peaks = [];
let lastVal = 0;
let lastTime = 0;

async function start() {
    try {
        const stream = await navigator.mediaDevices.getUserMedia({
            video: { facingMode: 'environment', width: 128, height: 128 }
        });
        video.srcObject = stream;
        
        // Flaş (Torch) kontrolü
        const track = stream.getVideoTracks()[0];
        if (track.getCapabilities().torch) {
            await track.applyConstraints({ advanced: [{ torch: true }] });
        }
        
        isActive = true;
        btn.innerText = "DURDUR";
        requestAnimationFrame(analyze);
    } catch (e) {
        alert("Kamera hatası: " + e.message);
    }
}

function analyze() {
    if (!isActive) return;

    // Küçük bir alanda yüksek yoğunluklu tarama
    bctx.drawImage(video, 0, 0, 50, 50);
    const data = bctx.getImageData(0, 0, 50, 50).data;
    
    let rSum = 0;
    for (let i = 0; i < data.length; i += 4) {
        // Kırmızı kanal yoğunluğu + yeşil kanaldan gelen paraziti azaltma
        rSum += data[i] - (data[i+1] * 0.2); 
    }
    
    const currentVal = rSum / (data.length / 4);
    
    // Hassas Türev Filtresi (Işık değişimlerini yakalar)
    const delta = currentVal - lastVal;
    lastVal = currentVal;

    // Sinyal listesine ekle (Normalizasyon)
    signal.push(delta);
    if (signal.length > 200) signal.shift();

    detectPulse(delta);
    drawWave();
    requestAnimationFrame(analyze);
}

function detectPulse(val) {
    const now = performance.now();
    // Peak tespiti: Belirli bir eşiğin üzerindeki ani yükselmeler
    if (val > 0.8 && (now - lastTime) > 450) { 
        if (lastTime > 0) {
            const rr = now - lastTime;
            peaks.push(rr);
            if (peaks.length > 15) peaks.shift();
            updateStats();
        }
        lastTime = now;
    }
}

function updateStats() {
    if (peaks.length < 5) return;

    const avgRR = peaks.reduce((a, b) => a + b) / peaks.length;
    const bpm = Math.round(60000 / avgRR);
    
    // HRV (Stres) Analizi: RMSSD yöntemi
    let diffs = 0;
    for(let i=1; i<peaks.length; i++) {
        diffs += Math.pow(peaks[i] - peaks[i-1], 2);
    }
    const rmssd = Math.sqrt(diffs / (peaks.length - 1));
    
    // Stres puanı (Ters orantı: düşük HRV = yüksek stres)
    let stress = Math.max(5, Math.min(95, 100 - (rmssd * 1.5)));

    document.getElementById('bpm').innerText = bpm;
    document.getElementById('stress').innerText = Math.round(stress);
}

function drawWave() {
    wctx.clearRect(0, 0, waveCanvas.width, waveCanvas.height);
    wctx.strokeStyle = '#ff3e3e';
    wctx.lineWidth = 3;
    wctx.lineJoin = 'round';
    wctx.beginPath();

    const middle = waveCanvas.height / 2;
    const step = waveCanvas.width / 200;

    for (let i = 0; i < signal.length; i++) {
        const x = i * step;
        const y = middle - (signal[i] * 15); // Hassasiyet çarpanı
        if (i === 0) wctx.moveTo(x, y);
        else wctx.lineTo(x, y);
    }
    wctx.stroke();
}

btn.addEventListener('click', () => {
    if (isActive) {
        isActive = false;
        btn.innerText = "ANALİZİ BAŞLAT";
        if (video.srcObject) video.srcObject.getTracks().forEach(t => t.stop());
    } else {
        start();
    }
});

// Canvas ölçekleme
waveCanvas.width = waveCanvas.offsetWidth;
waveCanvas.height = waveCanvas.offsetHeight;
</script>
